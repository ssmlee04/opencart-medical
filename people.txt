SELECT * FROM oc_product p LEFT JOIN oc_product_description pd ON (p.product_id = pd.product_id)
		LEFT JOIN oc_unit_class_description ucd ON (p.unit_class_id = ucd.unit_class_id AND ucd.language_id = pd.language_id) WHERE pd.language_id = '2' AND ( p.status = 1) AND ( product_type_id = 2) GROUP BY p.product_id ORDER BY pd.name ASC LIMIT 0,99999
